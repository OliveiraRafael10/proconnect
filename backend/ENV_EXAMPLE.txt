# Configurações do Supabase
# Obtenha essas informações no painel do Supabase: https://app.supabase.com

# URL do seu projeto Supabase (obrigatório)
SUPABASE_URL=https://seu-projeto.supabase.co

# Chave anônima (opcional, mas recomendada)
SUPABASE_ANON_KEY=sua-chave-anonima-aqui

# Chave de Service Role (obrigatório) - NUNCA exponha no frontend!
SUPABASE_SERVICE_ROLE_KEY=sua-chave-service-role-aqui

# Secret JWT do Supabase (obrigatório)
SUPABASE_JWT_SECRET=seu-jwt-secret-aqui

# Configurações de CORS (opcional)
CORS_ORIGINS=http://localhost:5173,http://localhost:3000

# Configurações de Cookies (opcionais)
COOKIE_DOMAIN=localhost
COOKIE_SECURE=false
COOKIE_SAMESITE=Lax
COOKIE_MAX_AGE=604800

# Host da API para Swagger (opcional)
API_HOST=localhost:5000
